Canvas

Canvas 基础

  • HTML 提供容器 Canvas(tag)，JavaScript API 提供接口来绘画.
  • Canvas 拥有绘制路径、矩形、圆形、字符以及添加图片等方法
  • Canvas 支持 2D, 3D(WebGL)

Example

HTML

指定 Canvas 画布尺寸. 


<canvas id="mycanvas" width="500px" height="500px">
  您的浏览器不支持 Canvas!
</canvas>

JavaScript


var canvas = document.getElementById( 'mycanvas' );
var context;
if ( canvas.getContext ) {
  context = canvas.getContext('2d'); // 使用 2D 渲染上下文
}

canvas基础知识 - 绘制路径

  • context.moveTo( x, y ); // 绘图游标移到(x, y)点, 但是不划线.
  • context.lineTo( x, y ); // 从上一点,开始绘制直线, 到 (x, y)为止.
  • context.stroke(); // 绘制路径
  • context.fill(); 

var canvas = document.getElementById( 'line' );
var context = canvas.getContext('2d');
context.beginPath();
context.lineWidth = 10;         // 设置画笔粗细(px)     
context.strokeStyle = 'green';  // 描边样式
context.lineCap = 'round';      // 设置线条末端形状 - butt/round/square
context.lineJoin = 'bevel';     // 设置线条相交方式 - round/bevel/miter
context.moveTo(10,10);
context.lineTo(100,100);
context.moveTo( 10, 100 );
context.lineTo( 100, 10 );
context.closePath();
context.stroke();

canvas基础知识 - 绘制弧度

context.arcTo( x1, y1, x2, y2, radius );
从上一点(x0, y0)开始绘制弧线, 到 (x2, y2) 为止, 并且以 radius 为半径, 同时与 (x0, y0), (x1, y1)线和(x1, y1), (x2, y2)线相切. 


var canvas = document.getElementById( 'arcto' );

// 以 50 为半径绘制 artTo.
context.moveTo( 0, 0 );
context.arcTo( 100, 0, 200, 200, 50 );
context.stroke();
// 画两条线做辅助.
context.strokeStyle = '#000';
context.moveTo( 0, 0 );
context.lineTo( 100, 0 );
context.lineTo( 200, 200 );
context.closePath();
// 绘制.
context.stroke();

canvas - 变形 - 矩阵

  • context.setTransform(1, 0, 0, 1, 0, 0);
  • context.transform( a, b, c, d, e, f );
    • a: X 轴缩放 cos
    • b: Y 轴倾斜 -sin
    • c: X 轴倾斜 sin
    • d: Y 轴缩放 cos
    • e: X 轴平移
    • f: Y 轴平移 

var canvas = document.getElementById( 'transform' );
var context = canvas.getContext( '2d' );
context.setTransform(1, 0, 0, 1, 0, 0);
var xScale = Math.cos( 0.7854 );
var ySkew = -Math.sin( 0.7854 );
var xSkew = Math.sin( 0.7854 );
var yScale = Math.cos( 0.7854 );
var xTransform = 200;
var yTransform = 200;
context.transform( xScale, ySkew, xSkew, yScale, xTransform, yTransform );
context.fillRect( -50, -50, 100, 100 );

canvas - globalCompositeOperation

图层合并方法. source: 当前正在绘制的路径; destination: 已绘制的路径;

  • source-over - Default. 后绘制的覆盖已绘制的.
  • destination-over. 已绘制的覆盖后绘制的.
  • source-atop 把source绘制在destination上, 重叠区域两者都不透明; 不重叠区域, source 透明, destination 不透明.
  • destination-atop 同 source-atop 相反.
  • source-in 绘制 source 和 destination 重叠部分的 source.
  • destination-int 同 source-in 相反, 绘制 source 和 destination 重叠部分的 destination.
  • source-out 在与 destination 不重复的区域绘制 source, 其它部分变成透明
  • destination-out 同 source-out 相反. 但设置透明度时, 重叠部分保留.
  • lighter 把 source 和 destination 重叠部分, 颜色相加, 最大值为 255;
  • copy 只绘制 source
  • xor 重叠区域透明. 

var canvas = document.getElementById( 'composite' );
var context = canvas.getContext( '2d' );
// @NOTE: 存在 alpha 透明.
// context.globalAlpha = 0.5;
context.fillStyle = 'green';
context.fillRect( 0, 0, 100, 100 );
context.fillStyle = '#faa';
context.globalCompositeOperation = 'xor';
context.fillRect( 50, 50, 100, 100);

canvas - 处理图像

  • context.drawImage( element, x, y[, width, height] );
    • element - img/canvas/video
    • x, y - 画布上绘制的起点坐标
    • width/height, 设置将要绘制的图像尺寸 - 会缩放图像
  • context.drawImage( image, sx, sy, sw, sh, dx, dy, dw, dh );
    • image - 图像
    • sx, sy - 图像源的 (x, y) 坐标
    • sw, sh - 裁剪图像的尺寸
    • dx, dy - 在画布上绘制的的起点坐标
    • dw, dh - 在画布上将要绘制的尺寸 

var canvas = document.getElementById( 'image' );
var context = canvas.getContext( '2d' );
// 加载图片
var image = new Image();
image.onload = function() {
  context.drawImage( image, 0, 200, 600, 600, 0, 0, 600, 600 );
};
image.src = './images/canvas/ouba.jpg';

canvas - 处理视频


var canvas = document.getElementById( 'video' );
var context = canvas.getContext( '2d' );
var video = document.getElementById( 'myvideo' );
canvas.onclick = function() {
  drawCanvas_reverse();
  video.play();
};
function drawCanvas_reverse(){
  if( video.paused || video.ended ) {
    return false;
  }
  context.drawImage(video,0,0,600,350);
  var imageData = context.getImageData(0,0,canvas.width,canvas.height);
  context.clearRect(0,0,canvas.width,canvas.height);
  var pixels = imageData.data;
  var numPixels = pixels.length;
  for(var i=0;i < numPixels;i++){
    pixels[i*4] = 255-pixels[i*4];
    pixels[i*4+1] = 255-pixels[i*4+1];
    pixels[i*4+2] = 255-pixels[i*4+2];
  }
  context.putImageData(imageData,0,0);
  setTimeout(function() {
    drawCanvas_reverse();
  }, 30);
}
