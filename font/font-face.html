@font-face {
  font-family: 'tt';
  /*
    IE 4 开始支持 @font-face.
    但仅支持 .eot 格式字体 - ms 提供压缩工具将 .ttf 转成 .eot.
    同时 IE 不支持 format 写法.
  */
  src: url('./font/tt.eot');
  src:
    url('./font/tt.eot?#font-spider') format('embedded-opentype'),
    /*
      woff
      内建了压缩功能，因而体积更小、方便下载和传播
      支持元信息数据
    */
    url('./font/tt.woff') format('woff'),
    url('./font/tt.ttf') format('truetype'),
    url('./font/tt.svg') format('svg');
  font-weight: normal;
  font-style: normal;
}

/*
    ## font-spider 限制：
    * 不支持行内样式
    * CSS content 属性插入的字符需要定义 font-family，不支持继承
    * 不支持 javascript 动态插入的样式与元素节点
    * 不支持 .otf 格式的字体
*/
